function upsertWeighted(items, type, delta = 1, keyName = "weight") {
  const found = items.find((x) => x.type === type);
  if (found) {
    found[keyName] = Math.min((found[keyName] || 0) + delta, 10);
  } else {
    items.push({ type, [keyName]: Math.min(delta, 10) });
  }
}

function sortWeighted(items, keyName = "weight") {
  items.sort((a, b) => (b[keyName] || 0) - (a[keyName] || 0));
}

function normalizeWeightedArray(arr, keyName = "weight") {
  if (!Array.isArray(arr)) return [];
  return arr
    .filter((x) => x && typeof x === "object" && typeof x.type === "string")
    .map((x) => ({
      type: x.type,
      [keyName]: typeof x[keyName] === "number" ? x[keyName] : 1,
    }));
}

function loadKnownUsers() {
  try {
    if (!fs.existsSync(USERS_FILE)) return [];

function normalizeMemoryShape(memory) {
  if (!memory || typeof memory !== "object") {
    return {
      name: "",
      likes: [],
      lastMessages: [],
      emotions: [],
      topics: [],
    };
  }

  if (!memory.name) memory.name = "";
  if (!Array.isArray(memory.likes)) memory.likes = [];
  if (!Array.isArray(memory.lastMessages)) memory.lastMessages = [];

  if (Array.isArray(memory.emotions)) {
    if (memory.emotions.length > 0 && typeof memory.emotions[0] === "string") {
      const converted = [];
      for (const e of memory.emotions) {
        upsertWeighted(converted, e, 1, "level");
      }
      memory.emotions = converted;
    } else {
      memory.emotions = normalizeWeightedArray(memory.emotions, "level");
    }
  } else {
    memory.emotions = [];
  }

  if (Array.isArray(memory.topics)) {
    if (memory.topics.length > 0 && typeof memory.topics[0] === "string") {
      const converted = [];
      for (const t of memory.topics) {
        upsertWeighted(converted, t, 1, "weight");
      }
      memory.topics = converted;
    } else {
      memory.topics = normalizeWeightedArray(memory.topics, "weight");
    }
  } else {
    memory.topics = [];
  }

  return memory;
}

function loadPersistentMemory() {
  try {
    if (!fs.existsSync(MEMORY_FILE)) return;

    const raw = fs.readFileSync(MEMORY_FILE, "utf8");
    const parsed = JSON.parse(raw);

    if (parsed.userModes) {
      for (const [k, v] of Object.entries(parsed.userModes)) {
        userModes.set(k, v);
      }
    }

    if (parsed.userMemory) {
      for (const [k, v] of Object.entries(parsed.userMemory)) {
        userMemory.set(k, normalizeMemoryShape(v));
      }
    }

    if (parsed.userLastSeen) {
      for (const [k, v] of Object.entries(parsed.userLastSeen)) {
        userLastSeen.set(k, v);
      }
    }
  } catch (err) {
    console.error("user-memory.json 読み込み失敗:", err.message);
  }
}

function savePersistentMemory() {
  try {
    const data = {
      userModes: Object.fromEntries(userModes),
      userMemory: Object.fromEntries(userMemory),
      userLastSeen: Object.fromEntries(userLastSeen),
    };
    fs.writeFileSync(MEMORY_FILE, JSON.stringify(data, null, 2), "utf8");
  } catch (err) {
    console.error("user-memory.json 保存失敗:", err.message);
  }
}

function persistUserData() {
  savePersistentMemory();
}

function cleanupOldUsers() {
  const now = Date.now();
  let changed = false;

  for (const [userId, ts] of userLastSeen.entries()) {
    if (now - ts > MEMORY_TTL_MS) {
      userConversations.delete(userId);
      userMemory.delete(userId);
      userLastSeen.delete(userId);
      userModes.delete(userId);
      changed = true;
    }
  }

  if (userLastSeen.size > MAX_USERS) {
    const sorted = [...userLastSeen.entries()].sort((a, b) => a[1] - b[1]);
    const deleteCount = userLastSeen.size - MAX_USERS;

    for (let i = 0; i < deleteCount; i++) {
      const [userId] = sorted[i];
      userLastSeen.delete(userId);
      userConversations.delete(userId);
      userMemory.delete(userId);
      userModes.delete(userId);
      changed = true;
    }
  }

  if (changed) persistUserData();
}

function updateLastSeen(userId) {
  userLastSeen.set(userId, Date.now());
  cleanupOldUsers();
  persistUserData();
}

function getUserMemory(userId) {

function extractName(text = "") {
  const patterns = [
    /(?:俺は|おれは|私は|わたしは|僕は|ぼくは)\s*([^\s、。]{1,20})/,
    /(?:名前は|なまえは)\s*([^\s、。]{1,20})/,
    /(?:俺の名前は|おれの名前は|私の名前は|わたしの名前は|僕の名前は|ぼくの名前は)\s*([^\s、。]{1,20})/,
  ];

  for (const pattern of patterns) {
    const match = text.match(pattern);
    if (match && match[1]) {
      return match[1].replace(/です|だよ|だ|です。|だよ。|だ。$/, "").trim();
    }
  }
  return "";
}

function extractLikes(text = "") {
  const likes = [];
  const patterns = [
    /(.+?)が好き/,
    /趣味は(.+?)(?:です|だ|。|、|$)/,
    /最近は(.+?)にはまってる/,
  ];

  for (const pattern of patterns) {
    const match = text.match(pattern);
    if (match && match[1]) {
      const value = match[1].trim();
      if (value && value.length <= 30) likes.push(value);
    }
  }

  return [...new Set(likes)];
}

function decayWeightedMemory(memory) {
  for (const e of memory.emotions) {
    e.level = Math.max((e.level || 1) - 0.1, 0.5);
  }
  for (const t of memory.topics) {
    t.weight = Math.max((t.weight || 1) - 0.05, 0.5);
  }
}

function trimWeightedMemory(memory) {
  sortWeighted(memory.emotions, "level");
  sortWeighted(memory.topics, "weight");
  memory.emotions = memory.emotions.slice(0, 10);
  memory.topics = memory.topics.slice(0, 10);
}

function rememberUserInfo(userId, text) {
  const memory = getUserMemory(userId);

  decayWeightedMemory(memory);

  const name = extractName(text);
  if (name) {
    memory.name = name;
  }

  const likes = extractLikes(text);
  for (const like of likes) {
    if (!memory.likes.includes(like)) {
      memory.likes.push(like);
    }
  }
  if (memory.likes.length > 10) {
    memory.likes = memory.likes.slice(-10);
  }

  const emotions = detectEmotionSignals(text);
  for (const emotion of emotions) {
    upsertWeighted(memory.emotions, emotion.type, emotion.level, "level");
  }

  const topics = detectTopicSignals(text);
  for (const topic of topics) {
    upsertWeighted(memory.topics, topic.type, topic.weight, "weight");
  }

  trimWeightedMemory(memory);
  updateLastMessages(userId, text);
  persistUserData();

  return {
    rememberedName: name || "",
    rememberedLikes: likes,
    rememberedEmotions: emotions,
    rememberedTopics: topics,
  };
}

function detectAutoMode(text = "") {
  const t = String(text).trim();

  if (/つらい|辛い|しんどい|苦しい|悲しい|泣きたい|消えたい|死にたい|もう無理|疲れた|限界|絶望|傷ついた|孤独|さみしい|寂しい/.test(t)) {
    return "obaachan";
  }

  if (/喝|叱って|厳しく|背中押して|はっきり|強めに|甘やかすな|渇を入れて|気合|動けるように/.test(t)) {
    return "babaa";
  }

  if (/どうすれば|どうしたら|整理|相談|考えたい|現実的|冷静に|アドバイス|判断|決めたい/.test(t)) {
    return "obaa";
  }

  return null;
}

function splitIntoLineMessages(text = "", maxMessages = 5) {
  const normalized = String(text)
    .replace(/\r\n/g, "\n")
    .replace(/\n{3,}/g, "\n\n")
    .trim();

  if (!normalized) return ["…"];

  const sentences = normalized
    .split(/(?<=[。！？\?!\n])/)
    .map((s) => s.trim())
    .filter(Boolean);

  const chunks = [];
  let current = "";

  for (const sentence of sentences) {
    if ((current + sentence).length <= 90) {
      current += sentence;
    } else {
      if (current) chunks.push(current.trim());
      current = sentence;
    }
  }

  if (current) chunks.push(current.trim());

  const finalChunks = [];
  for (const chunk of chunks) {
    if (chunk.length <= 120) {
      finalChunks.push(chunk);
    } else {
      for (let i = 0; i < chunk.length; i += 120) {
        finalChunks.push(chunk.slice(i, i + 120).trim());
      }
    }
  }

  return finalChunks.slice(0, maxMessages);
}

async function replyText(replyToken, text) {
  const messageTexts = splitIntoLineMessages(text, 5);

  await fetch("https://api.line.me/v2/bot/message/reply", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${LINE_CHANNEL_ACCESS_TOKEN}`,
    },
    body: JSON.stringify({
      replyToken,
      messages: messageTexts.map((t, index) => {
        const msg = { type: "text", text: t };
        if (index === messageTexts.length - 1) {
          msg.quickReply = buildQuickReply();
        }
        return msg;
      }),
    }),
  });
}

async function pushText(userId, text) {
  const messageTexts = splitIntoLineMessages(text, 5);

  await fetch("https://api.line.me/v2/bot/message/push", {
    method: "POST",
    headers: {
      "Content-Type": "application/json",
      Authorization: `Bearer ${LINE_CHANNEL_ACCESS_TOKEN}`,
    },
    body: JSON.stringify({
      to: userId,
      messages: messageTexts.map((t, index) => {
        const msg = { type: "text", text: t };
        if (index === messageTexts.length - 1) {
          msg.quickReply = buildQuickReply();
        }
        return msg;
      }),
    }),
  });
}

function verifyLineSignature(channelSecret, body, signature) {
  const hash = crypto
    .createHmac("SHA256", channelSecret)
    .update(body)
    .digest("base64");
  return hash === signature;
}

  const morningBase = [
    "おはよう。今日はひとつ出来たら十分だよ。",
    "おはようさん。朝はゆっくりでええよ。まず深呼吸しようか。",
    "おはよう。今日もぼちぼちでいこうね。",
  ];

  const nightBase = [
    "今日もよう頑張ったね。今夜は少し肩の力を抜きなさい。",
    "おつかれさん。全部片づいてなくても、今日はもう休んでええよ。",
    "夜は考えすぎんでよかよ。あったかくして休みなさい。",
  ];

  const pool = type === "night" ? nightBase : morningBase;
  let message = pool[Math.floor(Math.random() * pool.length)];

  if (type === "morning" && likeHint) {
    message += ` ${likeHint}のことでも、小さく一歩進めたら上出来だよ。`;
  }

  if (type === "morning" && topTopic) {
    message += ` ${topTopic}のことも、今日は抱えすぎんでええからね。`;
  }

  if (type === "night" && likeHint) {
    message += ` ${likeHint}のことは、また明日でもええからね。`;
  }

  if (type === "night" && topTopic) {
    message += ` ${topTopic}のことも、今夜は少し脇に置こうか。`;
  }

  if (type === "night" && /疲れている|つらい|不安|落ち込み/.test(topEmotion)) {
    message += " 今夜は無理に気合い入れんでええよ。";
  }

  if (name) {
    message = `${name}、${message}`;
  }

  return message;
}

function summarizeTopEmotions(emotions) {
  if (!emotions.length) return "相手の感情メモはまだ少ないです。";
  return emotions
    .slice(0, 3)
    .map((e) => `${e.type}(強さ${Math.round(e.level)})`)
    .join("、");
}

function summarizeTopTopics(topics) {
  if (!topics.length) return "相手の悩み分野メモはまだ少ないです。";
  return topics
    .slice(0, 3)
    .map((t) => `${t.type}(重要度${Math.round(t.weight)})`)
    .join("、");
}

function buildContinuityHint(memory) {
  if (!memory || !Array.isArray(memory.emotions) || !Array.isArray(memory.topics)) {
    return "継続している悩みのヒントはまだ少ないです。";
  }

  const strongEmotion = memory.emotions.find((e) => (e.level || 0) >= 5);
  const strongTopic = memory.topics.find((t) => (t.weight || 0) >= 5);

  if (strongEmotion && strongTopic) {
    return `相手は「${strongTopic.type}」のことで「${strongEmotion.type}」状態が続いている可能性があります。今回の返答では、単発の悩みではなく、しばらく続いている負担として自然ににじませてください。`;
  }

  if (strongEmotion) {
    return `相手は「${strongEmotion.type}」状態がしばらく続いている可能性があります。今回の返答では、一時的というより蓄積している感じを少しだけにじませてください。`;
  }

  if (strongTopic) {
    return `相手は「${strongTopic.type}」のテーマを繰り返し抱えている可能性があります。今回の返答では、そのテーマが継続案件である感じを少しだけにじませてください。`;
  }

  return "継続している悩みのヒントはまだ少ないです。";
}

async function generateAIReply(userId, userText) {
  const selectedMode = userModes.get(userId) || DEFAULT_MODE;
  const hasManualMode = userModes.has(userId);
  const autoMode = hasManualMode ? null : detectAutoMode(userText);
  const mode = autoMode || selectedMode;
  const systemPrompt = PROMPTS[mode] || PROMPTS.obaa;

  const memory = getUserMemory(userId);
  const conversation = getConversation(userId);

  const nameLine = memory.name
    ? `相手の名前は「${memory.name}」です。今回の返答では、できれば自然に1回は名前を入れてください。毎回連呼はしないでください。`
    : "相手の名前はまだ不明です。";

  const likesLine =
    memory.likes.length
      ? `相手が好きだと話したもの: ${memory.likes.join("、")}。話題が合う時は、1回だけ自然に触れてください。`
      : "相手の好み情報はまだ少ないです。";

  const emotionLine = `相手の最近の感情傾向: ${summarizeTopEmotions(memory.emotions)}。今回の返答では、必要ならその強さも踏まえて、寄り添い方や助言の強さを調整してください。`;

  const topicLine = `相手の最近の悩みや関心の分野: ${summarizeTopTopics(memory.topics)}。今回の返答では、必要ならどの分野の話かを理解している感じを少し出してください。`;

  const continuityLine = buildContinuityHint(memory);

  const recentLine =
    memory.lastMessages.length
      ? `相手の最近の発言: ${memory.lastMessages.join(" / ")}`
      : "最近の発言メモはまだ少ないです。";

  const convoInput = conversation.map((m) => ({
    role: m.role,
    content: m.content,
  }));

  const response = await client.responses.create({
    model: "gpt-5",
    input: [
      {
        role: "system",
        content: `${systemPrompt}

【全モード共通ルール】
- 返事は短めで自然に
- 1回の返答は基本1〜4文
- 毎回長文にしない
- AIっぽく説明しすぎない
- 相手が雑談なら雑談として返す
- 相手が深刻なら深刻さに合わせる
- LINEで読みやすいように、1文を短めにする
- 話題が変わるときは改行してよい
- 一度に詰め込みすぎない
- 日本語で返す
- 句読点は自然に使う
- 絶対に自分をAIだと前面に出さない
- 危険行為や自傷の具体的方法は案内しない
- 名前や好みや感情や話題分野を覚えている場合は、今回の返答で「覚えている感じ」が少し伝わるようにする
- ただし不自然に何個も詰め込まない

【相手について】
${nameLine}
${likesLine}
${emotionLine}
${topicLine}
${continuityLine}
${recentLine}`,
      },
      ...convoInput,
      {
        role: "user",
        content: userText,
      },
    ],
  });

  return (
    response.output_text?.trim() ||
    "ごめんね、ちょっと言葉がまとまらんかった。また言うてみて。"
  );
}

app.get("/push", async (req, res) => {
  try {
    const secret = req.query.secret || "";
    if (!PUSH_SECRET || secret !== PUSH_SECRET) {
      return res.status(403).send("forbidden");
    }

    const type = req.query.type === "night" ? "night" : "morning";
    const users = getKnownUsers();

    for (const userId of users) {
      const message = buildPushMessage(userId, type);
      await pushText(userId, message);
    }

    res.send(`ok: pushed ${users.length} users`);
  } catch (err) {
    console.error("/push error:", err);
    res.status(500).send("push failed");
  }
});

app.post(
  "/webhook",
  express.raw({ type: "application/json" }),
  async (req, res) => {
    try {
      const signature = req.headers["x-line-signature"];
      const body = req.body.toString("utf8");

      if (!verifyLineSignature(LINE_CHANNEL_SECRET, body, signature)) {
        console.error("署名検証NG");
        return res.status(401).send("invalid signature");
      }

      const json = JSON.parse(body);
      console.log("LINE webhook受信:", JSON.stringify(json, null, 2));

      const events = json.events || [];

      for (const event of events) {
        try {
          if (event.type !== "message" || event.message?.type !== "text") continue;

          const replyToken = event.replyToken;
          const userId = event.source?.userId;
          const userText = (event.message?.text || "").trim();

          if (!replyToken || !userId || !userText) continue;

          addKnownUser(userId);
          updateLastSeen(userId);

          if (userText === "おばあちゃん") {
            userModes.set(userId, "obaachan");
            persistUserData();
            await replyText(replyToken, getModeChangeMessage("obaachan"));
            continue;
          }

          if (userText === "おばあ") {
            userModes.set(userId, "obaa");
            persistUserData();
            await replyText(replyToken, getModeChangeMessage("obaa"));
            continue;
          }

          if (userText === "ばばあ") {
            userModes.set(userId, "babaa");
            persistUserData();
            await replyText(replyToken, getModeChangeMessage("babaa"));
            continue;
          }

          const selectedMode = userModes.get(userId) || DEFAULT_MODE;
          const hasManualMode = userModes.has(userId);
          const autoMode = hasManualMode ? null : detectAutoMode(userText);
          const mode = autoMode || selectedMode;

          const memoryResult = rememberUserInfo(userId, userText);
          const rememberedAck = buildRememberedAck(mode, memoryResult);

          if (rememberedAck) {
            await replyText(replyToken, rememberedAck);
            continue;
          }

          pushConversation(userId, "user", userText);

          const aiReply = await generateAIReply(userId, userText);

          pushConversation(userId, "assistant", aiReply);

          await replyText(replyToken, aiReply);
        } catch (eventErr) {
          console.error("イベント処理エラー:", eventErr);
          if (event.replyToken) {
            try {
              await replyText(
                event.replyToken,
                "ごめんね、ちょっと返事が詰まってしもうた。もう一回言うてみて。"
              );
            } catch (replyErr) {
              console.error("エラー返信失敗:", replyErr);
            }
          }
        }
      }

      res.sendStatus(200);
    } catch (err) {
      console.error("webhook 全体エラー:", err);
      res.status(500).send("error");
    }
  }
);
